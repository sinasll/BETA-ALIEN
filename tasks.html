// Function to set the profile data function setProfileData(name) { const usernameElement = document.getElementById('username'); usernameElement.innerText = name; } // Ensure the Telegram Web App SDK is ready window.Telegram.WebApp.ready(); // Check if SDK is ready and user data is available if (window.Telegram.WebApp.initDataUnsafe) { const user = window.Telegram.WebApp.initDataUnsafe.user; console.log('User Data:', user); // Log user data for debugging if (user) { const name = user.username ? `@${user.username}` : user.first_name || 'Unknown User'; setProfileData(name); } else { console.error('User information is not available'); setProfileData('Unknown User'); // Default fallback in case of error } } else { console.error('Telegram Web App SDK initialization failed'); setProfileData('Unknown User'); // Default fallback if SDK is not initialized }
